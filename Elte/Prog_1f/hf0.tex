\vspace{5mm} \\
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\fejezet{Specifikáció}
Az értékeket egy tömbben tároljuk.
\begin{flalign*}
&A=(t:\mathbb{R}^n,~max:\mathbb{R},~ind:\mathbb{N},~l:\mathbb{L})\\
&Ef=(t=t')\\
&Uf=(Ef~\wedge~\max_{i=2,~t[i]<t[i-1],~t[i]<t[i+1]}^{i=n-1}=t[i])\\
\end{flalign*}
